            text = text.split('[')[0]
            cell = None

                cell = text.split('')[-1].strip()
                
                try:
                    cell = datetime.strptime(cell, '%d %b %Y').date()
                except ValueError:
                    cell = '15 ' + cell
                    cell = datetime.strptime(cell, '%d %b %Y').date()
                
            elif i == 2:
                cell = ('').join(text.split(','))
                try:
                    cell = int(cell)
                except ValueError:
                    cell = np.nan 
            elif i > 1:
                try:
                    cell = float(text)
                except ValueError:
                    cell = np.nan
            else:
                cell = text
                
            if i == len(tds)-1 and cell == np.nan:
                cell = 0.0

            x = []
            y = []

            for i in range(len(df[party])):
                if not np.isnan(df[party][i]):

                    y.append(df[party][i])

            x = np.array(x)
            y = np.array(y)

            # extrapolation
            election_day = datetime(2020, 10, 17).date()


            epoch_float = (datetime.combine(epoch, datetime.min.time())-datetime(1970,1,1)).total_seconds()

            loess = Loess(x, y)

            xx = np.linspace(min(df['Date_float']), (election_day - epoch).total_seconds(), 30)
            yy = np.zeros_like(xx)

            for i in range(len(xx)):
                yy[i] = loess.estimate(xx[i], window=window)



    #         p = np.polyfit(xx, yy, 10)

            xxx = np.linspace(min(df['Date_float']), (election_day - epoch).total_seconds(), 1000)
